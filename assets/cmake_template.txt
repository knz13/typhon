project(__PROJECT__NAME__)

    #__PROJECT__SOURCES__

)

    C_STANDARD 20
)

link_directories(__TYPHON__LIBRARY__LOCATION__) #__TYPHON__LIBRARY_LOCATION__LINE__

target_link_libraries(${PROJECT_NAME} PUBLIC typhon)
